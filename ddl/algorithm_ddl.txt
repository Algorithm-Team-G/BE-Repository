drop database algorithm_project;
create database algorithm_project;
use algorithm_project;

create table T_team (
	team_id INT primary key,
    team_name varchar(30)
);

create table T_job (
	job_id INT primary key,
    job_name varchar(30),
    team_id INT,
    foreign key (team_id) references T_team (team_id)
);

create table T_worker (
	worker_id INT primary key,
    worker_name varchar(20),
    job_id INT,
    career INT,
    max_task_count INT,
    foreign key (job_id) references T_job(job_id)
);

create table T_task (
    task_id INT AUTO_INCREMENT primary key,
    task_name varchar(200),
    start_date timestamp,
    end_date timestamp,
    level INT,
    importance INT,
    job_id INT,
    worker_id INT,
    foreign key (worker_id) references T_worker (worker_id)
);

create view t_worker_task_count as (
	select t.worker_id, count(*) count
    from t_task t
    group by t.worker_id
    having t.worker_id is not null
);
